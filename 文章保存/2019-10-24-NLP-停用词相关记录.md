```python
def get_stopwords():
    path = Myconfig.get_path('stopwords.txt')
    stopwords = []
    with open(path) as f:
        line = f.readline()
        while line != '':
            stopwords.append(line.strip('\n'))
            line = f.readline()
	# 去掉文本中的全角空白符，不间断空白符 &nbsp;，字节顺序标记，换行
    stopwords += [' ','\u3000','\xa0','\ufeff','\t','\n','\\n']
    return set(stopwords)
	
```
